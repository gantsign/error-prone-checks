52  /**
53   * Compare a file transformed as suggested by {@link BugChecker} to an expected source.
54   *
55   * <p>Inputs are a {@link BugChecker} instance, input file and expected file.
56   *
57   * @author kurs@google.com (Jan Kurs)
58   */
59  public class BugCheckerRefactoringTestHelper {
60  
61    /** Test mode for matching refactored source against expected source. */
62    public enum TestMode {
63      TEXT_MATCH {
64        @Override
65        void verifyMatch(JavaFileObject refactoredSource, JavaFileObject expectedSource)
66            throws IOException {
67          assertThat(refactoredSource.getCharContent(false).toString())
68              .isEqualTo(expectedSource.getCharContent(false).toString());
69        }
70      },
71      AST_MATCH {
72        @Override
73        void verifyMatch(JavaFileObject refactoredSource, JavaFileObject expectedSource) {
74          assertAbout(javaSource()).that(refactoredSource).parsesAs(expectedSource);
75        }
76      };
77  
78      abstract void verifyMatch(JavaFileObject refactoredSource, JavaFileObject expectedSource)
79          throws IOException;
80    }
81  
82    /**
83     * For checks that provide multiple possible fixes, chooses the one that will be applied for the
84     * test.
85     */
86    public interface FixChooser {
87      Fix choose(List<Fix> fixes);
88    }
89  
90    enum FixChoosers implements FixChooser {
91      FIRST {
92        @Override
93        public Fix choose(List<Fix> fixes) {
94          return fixes.get(0);
95        }
96      }
97    }
98  
99    private final Map<JavaFileObject, JavaFileObject> sources = new HashMap<>();
100   private final BugChecker refactoringBugChecker;
101   private final ErrorProneInMemoryFileManager fileManager;
102 
103   private FixChooser fixChooser = FixChoosers.FIRST;
104   private List<String> options = ImmutableList.of();
105 
106   private BugCheckerRefactoringTestHelper(BugChecker refactoringBugChecker, Class<?> clazz) {
107     this.refactoringBugChecker = refactoringBugChecker;
108     this.fileManager = new ErrorProneInMemoryFileManager(clazz);
109   }
110 
111   public static BugCheckerRefactoringTestHelper newInstance(
112       BugChecker refactoringBugChecker, Class<?> clazz) {
113     return new BugCheckerRefactoringTestHelper(refactoringBugChecker, clazz);
114   }
115 
116   public BugCheckerRefactoringTestHelper.ExpectOutput addInput(String inputFilename) {
117     return new ExpectOutput(fileManager.forResource(inputFilename));
118   }
119 
120   public BugCheckerRefactoringTestHelper.ExpectOutput addInputLines(String path, String... input) {
121     assertThat(fileManager.exists(path)).isFalse();
122     return new ExpectOutput(fileManager.forSourceLines(path, input));
123   }
124 
125   public BugCheckerRefactoringTestHelper setFixChooser(FixChooser chooser) {
126     this.fixChooser = chooser;
127     return this;
128   }
129 
130   public BugCheckerRefactoringTestHelper setArgs(String... args) {
131     this.options = ImmutableList.copyOf(args);
132     return this;
133   }
134 
135   public void doTest() throws IOException {
136     this.doTest(TestMode.AST_MATCH);
137   }
138 
139   public void doTest(TestMode testMode) throws IOException {
140     for (Map.Entry<JavaFileObject, JavaFileObject> entry : sources.entrySet()) {
141       runTestOnPair(entry.getKey(), entry.getValue(), testMode);
142     }
143   }
144 
145   private BugCheckerRefactoringTestHelper addInputAndOutput(
146       JavaFileObject input, JavaFileObject output) {
147     sources.put(input, output);
148     return this;
149   }
150 
151   private void runTestOnPair(JavaFileObject input, JavaFileObject output, TestMode testMode)
152       throws IOException {
153     JavacTool tool = JavacTool.create();
154     DiagnosticCollector<JavaFileObject> diagnosticsCollector = new DiagnosticCollector<>();
155     Context context = new Context();
156     context.put(ErrorProneOptions.class, ErrorProneOptions.empty());
157     JavacTaskImpl task =
158         (JavacTaskImpl)
159             tool.getTask(
160                 CharStreams.nullWriter(),
161                 fileManager,
162                 diagnosticsCollector,
163                 options,
164                 /*classes=*/ null,
165                 ImmutableList.copyOf(sources.keySet()),
166                 context);
167     JCCompilationUnit tree = parseAndAnalyze(task, input);
168     Iterable<Diagnostic<? extends JavaFileObject>> errorDiagnostics =
169         Iterables.filter(
170             diagnosticsCollector.getDiagnostics(),
171             new Predicate<Diagnostic<? extends JavaFileObject>>() {
172               @Override
173               public boolean apply(Diagnostic<? extends JavaFileObject> input) {
174                 return input.getKind() == Diagnostic.Kind.ERROR;
175               }
176             });
177     if (!Iterables.isEmpty(errorDiagnostics)) {
178       fail("compilation failed unexpectedly: " + errorDiagnostics);
179     }
180     JavaFileObject transformed = applyDiff(input, context, tree);
181     testMode.verifyMatch(transformed, output);
182   }
183 
184   private JavaFileObject applyDiff(
185       JavaFileObject sourceFileObject, Context context, JCCompilationUnit tree) throws IOException {
186     final DescriptionBasedDiff diff = DescriptionBasedDiff.create(tree);
187     transformer(refactoringBugChecker)
188         .apply(
189             new TreePath(tree),
190             context,
191             new DescriptionListener() {
192               @Override
193               public void onDescribed(Description description) {
194                 if (!description.fixes.isEmpty()) {
195                   diff.handleFix(fixChooser.choose(description.fixes));
196                 }
197               }
198             });
199     SourceFile sourceFile = SourceFile.create(sourceFileObject);
200     diff.applyDifferences(sourceFile);
201 
202     JavaFileObject transformed =
203         JavaFileObjects.forSourceString(
204             Iterables.getOnlyElement(Iterables.filter(tree.getTypeDecls(), JCClassDecl.class))
205                 .sym
206                 .getQualifiedName()
207                 .toString(),
208             sourceFile.getSourceText());
209     return transformed;
210   }
211 
212   private JCCompilationUnit parseAndAnalyze(JavacTaskImpl task, final JavaFileObject input) {
213     Iterable<? extends CompilationUnitTree> trees = task.parse();
214     task.analyze();
215     return Iterables.getOnlyElement(
216         Iterables.filter(
217             Iterables.filter(trees, JCCompilationUnit.class),
218             new Predicate<JCCompilationUnit>() {
219               @Override
220               public boolean apply(JCCompilationUnit compilation) {
221                 return compilation.getSourceFile() == input;
222               }
223             }));
224   }
225 
226   private ErrorProneScannerTransformer transformer(BugChecker bugChecker) {
227     ErrorProneScanner scanner = new ErrorProneScanner(bugChecker);
228     return ErrorProneScannerTransformer.create(scanner);
229   }

231   /** To assert the proper {@code .addInput().addOutput()} chain. */
232   public class ExpectOutput {
233     private final JavaFileObject input;
234 
235     public ExpectOutput(JavaFileObject input) {
236       this.input = input;
237     }
238 
239     public BugCheckerRefactoringTestHelper addOutputLines(String path, String... output)
240         throws IOException {
241       if (fileManager.exists(path)) {
242         throw new FileAlreadyExistsException(path);
243       }
244       return addInputAndOutput(input, fileManager.forSourceLines(path, output));
245     }
246 
247     public BugCheckerRefactoringTestHelper addOutput(String outputFilename) {
248       return addInputAndOutput(input, fileManager.forResource(outputFilename));
249     }
250 
251     public BugCheckerRefactoringTestHelper expectUnchanged() {
252       return addInputAndOutput(input, input);
253     }
254   }
