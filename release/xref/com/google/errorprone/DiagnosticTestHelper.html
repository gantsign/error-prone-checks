78    public static Matcher<Diagnostic<? extends JavaFileObject>> suggestsRemovalOfLine(
79        URI fileURI, int line) {
80      return allOf(
81          diagnosticOnLine(fileURI, line),
82          diagnosticMessage(containsString("remove this line")));
83    }

93    public String describe() {
94      StringBuilder stringBuilder = new StringBuilder().append("Diagnostics:\n");
95      for (Diagnostic<? extends JavaFileObject> diagnostic : getDiagnostics()) {
96        stringBuilder.append("  [")
97            .append(diagnostic.getLineNumber()).append(":")
98            .append(diagnostic.getColumnNumber())
99            .append("]\t");
100       stringBuilder.append(diagnostic.getMessage(Locale.getDefault()).replaceAll("\n", "\\\\n"));
101       stringBuilder.append("\n");
102     }
103     return stringBuilder.toString();
104   }

106   public static Matcher<Diagnostic<? extends JavaFileObject>> diagnosticLineAndColumn(
107       final long line, final long column) {
108     return new TypeSafeDiagnosingMatcher<Diagnostic<? extends JavaFileObject>>() {
109       @Override
110       protected boolean matchesSafely(Diagnostic<? extends JavaFileObject> item,
111           Description mismatchDescription) {
112         if (item.getLineNumber() != line) {
113           mismatchDescription.appendText("diagnostic not on line ")
114               .appendValue(item.getLineNumber());
115           return false;
116         }
117 
118         if (item.getColumnNumber() != column) {
119           mismatchDescription.appendText("diagnostic not on column ")
120               .appendValue(item.getColumnNumber());
121           return false;
122         }
123 
124         return true;
125       }
126 
127       @Override
128       public void describeTo(Description description) {
129         description
130             .appendText("a diagnostic on line:column ")
131             .appendValue(line)
132             .appendText(":")
133             .appendValue(column);
134       }
135     };
136   }

138   public static Matcher<Diagnostic<? extends JavaFileObject>> diagnosticOnLine(
139       final URI fileURI, final long line) {
140     return new TypeSafeDiagnosingMatcher<Diagnostic<? extends JavaFileObject>>() {
141       @Override
142       public boolean matchesSafely(Diagnostic<? extends JavaFileObject> item,
143           Description mismatchDescription) {
144         if (item.getSource() == null) {
145           mismatchDescription.appendText("diagnostic not attached to a file: ")
146               .appendValue(item.getMessage(ENGLISH));
147           return false;
148         }
149 
150         if (!item.getSource().toUri().equals(fileURI)) {
151           mismatchDescription.appendText("diagnostic not in file ").appendValue(fileURI);
152           return false;
153         }
154 
155         if (item.getLineNumber() != line) {
156           mismatchDescription.appendText("diagnostic not on line ")
157               .appendValue(item.getLineNumber());
158           return false;
159         }
160 
161         return true;
162       }
163 
164       @Override
165       public void describeTo(Description description) {
166         description
167             .appendText("a diagnostic on line ")
168             .appendValue(line);
169       }
170     };
171   }

173   public static Matcher<Diagnostic<? extends JavaFileObject>> diagnosticOnLine(
174       final URI fileURI, final long line, final Predicate<? super String> matcher) {
175     return new TypeSafeDiagnosingMatcher<Diagnostic<? extends JavaFileObject>>() {
176       @Override
177       public boolean matchesSafely(Diagnostic<? extends JavaFileObject> item,
178           Description mismatchDescription) {
179         if (item.getSource() == null) {
180           mismatchDescription.appendText("diagnostic not attached to a file: ")
181               .appendValue(item.getMessage(ENGLISH));
182           return false;
183         }
184 
185         if (!item.getSource().toUri().equals(fileURI)) {
186           mismatchDescription.appendText("diagnostic not in file ").appendValue(fileURI);
187           return false;
188         }
189 
190         if (item.getLineNumber() != line) {
191           mismatchDescription.appendText("diagnostic not on line ")
192               .appendValue(item.getLineNumber());
193           return false;
194         }
195 
196         if (!matcher.apply(item.getMessage(Locale.getDefault()))) {
197           mismatchDescription.appendText("diagnostic does not match ").appendValue(matcher);
198           return false;
199         }
200 
201         return true;
202       }
203 
204       @Override
205       public void describeTo(Description description) {
206         description
207             .appendText("a diagnostic on line ")
208             .appendValue(line)
209             .appendText(" that matches \n")
210             .appendValue(matcher)
211             .appendText("\n");
212       }
213     };
214   }

217   public static Matcher<Diagnostic<? extends JavaFileObject>> diagnosticMessage(
218       final Matcher<String> matcher) {
219     return new TypeSafeDiagnosingMatcher<Diagnostic<? extends JavaFileObject>>() {
220       @Override
221       public boolean matchesSafely(Diagnostic<? extends JavaFileObject> item,
222           Description mismatchDescription) {
223         if (!matcher.matches(item.getMessage(Locale.getDefault()))) {
224           mismatchDescription.appendText("diagnostic message does not match ")
225               .appendDescriptionOf(matcher);
226           return false;
227         }
228 
229         return true;
230       }
231 
232       @Override
233       public void describeTo(Description description) {
234         description.appendText("a diagnostic with message ").appendDescriptionOf(matcher);
235       }
236     };
237   }

239   /**
240    * Comment that marks a bug on the next line in a test file.  For example,
241    * "// BUG: Diagnostic contains: foo.bar()", where "foo.bar()" is a string that should be in the
242    * diagnostic for the line.  Multiple expected strings may be separated by newlines, e.g.
243    * // BUG: Diagnostic contains: foo.bar()
244    * // bar.baz()
245    * // baz.foo()
246    */
247   private static final String BUG_MARKER_COMMENT_INLINE = "// BUG: Diagnostic contains:";
248   private static final String BUG_MARKER_COMMENT_LOOKUP = "// BUG: Diagnostic matches:";
249   private final Set<String> usedLookupKeys = new HashSet<>();
250 
251   enum LookForCheckNameInDiagnostic {
252     YES,
253     NO;
254   }
255   
256   /**
257    * Expects an error message matching {@code matcher} at the line below a comment matching the key.
258    * For example, given the source
259    * <pre>
260    *   // BUG: Diagnostic matches: X
261    *   a = b + c;
262    * </pre>
263    * ... you can use
264    * {@code expectErrorMessage("X", Predicates.containsPattern("Can't add b to c"));}
265    *
266    * <p>Error message keys that don't match any diagnostics will cause test to fail.
267    */
268   public void expectErrorMessage(String key, Predicate<? super String> matcher) {
269     expectedErrorMsgs.put(key, matcher);
270   }
271 
272   /**
273    * Asserts that the diagnostics contain a diagnostic on each line of the source file that
274    * matches our bug marker pattern.  Parses the bug marker pattern for the specific string to
275    * look for in the diagnostic.
276    * @param source File in which to find matching lines
277    *
278    * TODO(eaftan): Switch to use assertThat instead of assertTrue.
279    */
280   public void assertHasDiagnosticOnAllMatchingLines(
281       JavaFileObject source, LookForCheckNameInDiagnostic lookForCheckNameInDiagnostic)
282       throws IOException {
283     final List<Diagnostic<? extends JavaFileObject>> diagnostics = getDiagnostics();
284     final LineNumberReader reader = new LineNumberReader(
285         CharSource.wrap(source.getCharContent(false)).openStream());
286     do {
287       String line = reader.readLine();
288       if (line == null) {
289         break;
290       }
291 
292       List<Predicate<? super String>> predicates = null;
293       if (line.contains(BUG_MARKER_COMMENT_INLINE)) {
294         // Diagnostic must contain all patterns from the bug marker comment.
295         List<String> patterns = extractPatterns(line, reader, BUG_MARKER_COMMENT_INLINE);
296         predicates = new ArrayList<>(patterns.size());
297         for (String pattern : patterns) {
298           predicates.add(new SimpleStringContains(pattern));
299         }
300       } else if (line.contains(BUG_MARKER_COMMENT_LOOKUP)) {
301         int markerLineNumber = reader.getLineNumber();
302         List<String> lookupKeys = extractPatterns(line, reader, BUG_MARKER_COMMENT_LOOKUP);
303         predicates = new ArrayList<>(lookupKeys.size());
304         for (String lookupKey : lookupKeys) {
305           assertTrue(
306               "No expected error message with key [" + lookupKey + "] as expected from line ["
307               + markerLineNumber + "] with diagnostic [" + line.trim() + "]",
308               expectedErrorMsgs.containsKey(lookupKey));
309           predicates.add(expectedErrorMsgs.get(lookupKey));
310           usedLookupKeys.add(lookupKey);
311         }
312       }
313       
314       if (predicates != null) {
315         int lineNumber = reader.getLineNumber();
316         for (Predicate<? super String> predicate : predicates) {
317           Matcher<? super Iterable<Diagnostic<? extends JavaFileObject>>> patternMatcher =
318               hasItem(diagnosticOnLine(source.toUri(), lineNumber, predicate));
319           assertTrue(
320               "Did not see an error on line " + lineNumber + " matching " + predicate
321                   + ". All errors:\n" + diagnostics,
322               patternMatcher.matches(diagnostics));
323         }
324 
325         if (checkName != null && lookForCheckNameInDiagnostic == LookForCheckNameInDiagnostic.YES) {
326           // Diagnostic must contain check name.
327           Matcher<? super Iterable<Diagnostic<? extends JavaFileObject>>> checkNameMatcher =
328               hasItem(diagnosticOnLine(
329                   source.toUri(), lineNumber, new SimpleStringContains("[" + checkName + "]")));
330           assertTrue(
331               "Did not see an error on line " + lineNumber + " containing [" + checkName
332                   + "]. All errors:\n" + diagnostics,
333               checkNameMatcher.matches(diagnostics));
334         }
335 
336       } else {
337         int lineNumber = reader.getLineNumber();
338         Matcher<? super Iterable<Diagnostic<? extends JavaFileObject>>> matcher =
339             hasItem(diagnosticOnLine(source.toUri(), lineNumber));
340         if (matcher.matches(diagnostics)) {
341           fail("Saw unexpected error on line " + lineNumber + ". All errors:\n" + diagnostics);
342         }
343       }
344     } while (true);
345     reader.close();
346   }

348   /** Returns the lookup keys that weren't used. */
349   public Set<String> getUnusedLookupKeys() {    
350     return Sets.difference(expectedErrorMsgs.keySet(), usedLookupKeys);
351   }

353   /**
354    * Extracts the patterns from a bug marker comment.
355    *
356    * @param line The first line of the bug marker comment
357    * @param reader A reader for the test file
358    * @param matchString The bug marker comment match string.
359    * @return A list of patterns that the diagnostic is expected to contain
360    * @throws IOException
361    */
362   private static List<String> extractPatterns(
363       String line, BufferedReader reader, String matchString)
364       throws IOException {
365     int bugMarkerIndex = line.indexOf(matchString);
366     if (bugMarkerIndex < 0) {
367       throw new IllegalArgumentException("Line must contain bug marker prefix");
368     }
369     List<String> result = new ArrayList<String>();
370     String restOfLine = line.substring(bugMarkerIndex + matchString.length()).trim();
371     result.add(restOfLine);
372     line = reader.readLine().trim();
373     while (line.startsWith("//")) {
374       restOfLine = line.substring(2).trim();
375       result.add(restOfLine);
376       line = reader.readLine().trim();
377     }
378 
379     return result;
380   }

382   private static class SimpleStringContains implements Predicate<String>  {
383     private final String pattern;
384 
385     SimpleStringContains(String pattern) {
386       this.pattern = pattern;
387     }
388     
389     @Override
390     public boolean apply(String input) {
391       return input.contains(pattern);
392     }
393     
394     @Override
395     public String toString() {
396       return pattern;
397     }
398   }

400   private static class ClearableDiagnosticCollector<S> implements DiagnosticListener<S> {
401     private final List<Diagnostic<? extends S>> diagnostics = new ArrayList<>();
402 
403     @Override
404     public void report(Diagnostic<? extends S> diagnostic) {
405       diagnostics.add(diagnostic);
406     }
407 
408     public List<Diagnostic<? extends S>> getDiagnostics() {
409       return ImmutableList.copyOf(diagnostics);
410     }
411 
412     public void clear() {
413       diagnostics.clear();
414     }
